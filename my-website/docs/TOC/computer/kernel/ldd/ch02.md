# 第二章 构造和运行模块

## Hello World

`hello.c`:

```c
/* the below headers are required for all modules */
#include <linux/init.h>
#include <linux/module.h>

/* struct task_struct for current */
#include <linux/sched.h>

/* License statement */
MODULE_LICENSE("GPL");

/* init function */
static int hello_init(void)
{
	/* $current is the pointer of this process */
	printk(KERN_INFO "process name: %s\n", current->comm);

	printk(KERN_ALERT "Hello, world\n");
	return 0;
}

/* cleanup function */
static void hello_exit(void)
{
	printk(KERN_ALERT "Gooldbye, cruel world\n");
}

/* init entry */
module_init(hello_init);

/* cleanup entry */
module_exit(hello_exit);
```

`Makefile`:

```makefile
ifneq ($(KERNELRELEASE),)
	obj-m := hello.o
else
	KERN_DIR ?= /lib/modules/$(shell uname -r)/build
	PWD := $(shell pwd)
default:
	$(MAKE) -C $(KERN_DIR) M=$(PWD) modules
endif

clean:
	rm -rf *.o *~ core .depend .*.cmd *.ko *.mod.c .tmp_versions
```
装载与使用:

```sh
# 装载
sudo insmod hello.ko

# 卸载
sudo rmmod hello

# 查看输出
sudo dmesg -T | grep -A 5 hello
```
